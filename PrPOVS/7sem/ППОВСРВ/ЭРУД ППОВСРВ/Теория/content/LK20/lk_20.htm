m-Law – используется в США, Японии;
A-Law – принят в Европе.

Оба метода дают примерно одинаковую картинку и заметные отличия начинаются с уровня 1/32. При этом первый метод дает несколько завышенные значения относительно логарифмической шкалы что позволяет перекрыть несколько больший диапазон входного сигнала (до 13 бит после запятой), тогда как предел для второго метода – 12 бит.
При кодировании формируется восьмибитное значение следующего формата Y = PSSSQQQQ, где P – бит знака (0 – плюс, 1 – минус), SSS – трехбитный номер сегмента, QQQQ – четырехбитный уровень квантования.
В процессе передачи полученные значения кода могут инвертироваться целиком или четные/нечетные биты.

Программная реализация метода μ-255

Ниже представлены примеры программ сжатия и восстановления данных по методу μ-255.

; Сжатие и восстановление сигналов методом m-law

        
; ППО ЭВС РВ Лекции
; (c) Клюс В.Б. 08/12/04 07:00
	.mmregs				; Определение регистров
TEMP	.set	60h
ONE	.set	61h
X	.set	62h
Y	.set	63h
P	.set	64h
Z	.set	65h
Cnt	.set	2048-1			; Число элементов массива
Stp	.set	32				; Шаг приращения
Buf	.set	1000h				; Буфер ввода
BufI	.set	2000h				; Буфер ввода
BufO	.set	3000h				; Буфер вывода
	.def	_c_int00			; Метка начала программы
	.align	80h			; Выровнять на начало страницы
_c_int00:
	stm	#BufI, SP			; Указатель стека
	ld	#0, DP			; Текущая страница
	st	#1, ONE
	ssbx	SXM				; Расширение знака
	ssbx	OVM				; 
	stm	#Cnt, BRC
	ld	#8000h-Stp, a
	rptbd	Fill-1
	stm	#BufI, ar1			;2w

	add	#Stp, a
	stl	a, *ar1+
Fill:
	stm	#Cnt, ar1
	stm	#BufI, ar2
	stm	#BufO, ar3
	stm	#Buf, ar4
Loop:
	mvdk	*ar2+, X
;	stm	#-0aaah, X
	call	MlawCom
	mvkd	Y, *ar3+
	call	MlawExp
	call	MlawTab
	mvkd	X, *ar4+
	mvkd	Z, *ar4(Cnt)
	banz	Loop, *ar1-
	nop
	nop

MlawCom:					; Normal -> m-law
;	portr	PA0, X
;;	stm	#4, X
	ld	X, 16, a
	bcd	ts1, ageq
;	bcd	ts1, aleq			; для таблицы
	stm	#0, P				;2w

	orm	#8000h, P			;
ts1:	abs	a
	rsbx	OVA
	add	#210h, 14, a		;+33
	bcd	ts2, aov
	exp	a				;1w
	ld	#7, b				;1w

	sub	T, b
	addm	#-6, T
	ld	bl, 16, b
	norm	a
	and	#0F0h, 16, a
	or	b, 8, a
ts2:
	sfta	a, 4
	add	P, 16, a
	cmpl	a
	and	#0FF00h, 16, a
	sth	a, Y
;	portw	Y, PA0
	ret
MlawExp:					; m-law -> Normal
;	portr	PA0, Y
	ld	Y, 16, a
	cmpl	a
	bcd	tc1, ageq
	stm	#0, P				;2w

	orm	#8000h, P			;
tc1:	ld	a, -4, b
	and	#7000h, 16, a		; S
	and	#0F0h, 16, b

	add	#-1000h, 16, a
	sth	a, -12, T
	add	#108h, 16, b		; +33
	norm	b
tc2:
	bitf	P, #8000h
	bcd	tc3, ntc
	add	#-108h, 15, b		;-33
	neg	b
tc3:
	sth	b, X
;	portw	X, PA0
	ret
MlawTab:					; A-law -> Normal
	ld	Y, 8, a
;	cmpl	a
	and	#0FFh, 16, a
	add	#MLAW, 16, a
	sth	a, ar5
	nop
	nop
	ld	*ar5, 2, b
	stl	b, Z
	ret
	.align	80h			; Выровнять на начало страницы
	.include "M-law.inc"
	.end					; Конец программы
